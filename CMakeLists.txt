project (fpd)

find_package(Threads)

if (UNIX)
 find_package(OpenSSL REQUIRED)
endif (UNIX)

ADD_EXECUTABLE(fpd fpd.cpp)

SET_PROPERTY(TARGET fpd PROPERTY CXX_STANDARD 17)

TARGET_INCLUDE_DIRECTORIES(fpd PRIVATE INCLUDE)

TARGET_INCLUDE_DIRECTORIES(fpd PUBLIC .)

if (WIN32)
  TARGET_LINK_LIBRARIES(fpd Ws2_32.lib)
else()
  TARGET_LINK_LIBRARIES(fpd ${CMAKE_THREAD_LIBS_INIT} OpenSSL::SSL)
endif (WIN32)
